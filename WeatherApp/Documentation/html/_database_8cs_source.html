224 // Load all maintenance records and parse dates, emails
225 public List<MaintenanceTask> GetMaintenanceData()
226 {
227 List<MaintenanceTask> maintenanceTasks = new List<MaintenanceTask>();
228
229 using (SqlConnection connection = new SqlConnection(_connectionString))
230 {
231 connection.Open();
232 string query = "SELECT m.sensor_id, s.sensor_type, m.maintenance_date, m.maintainer_id, m.maintainer_comments " +
233 "FROM dbo.Maintenance m " +
234 "INNER JOIN dbo.Sensors s ON m.sensor_id = s.sensor_id";
235
236 SqlCommand command = new SqlCommand(query, connection);
237
238 using (SqlDataReader reader = command.ExecuteReader())
239 {
240 while (reader.Read())
241 {
242 var task = new MaintenanceTask
243 {
244 SensorId = reader.GetInt32(reader.GetOrdinal("sensor_id")),
245 SensorType = reader.GetString(reader.GetOrdinal("sensor_type")),
246 MaintainerComments = reader.IsDBNull(reader.GetOrdinal("maintainer_comments"))
247 ? "No comments"
248 : reader.GetString(reader.GetOrdinal("maintainer_comments")),
249 Status = "Scheduled"
250 };
251
252 // Handle various date formats
253 if (!reader.IsDBNull(reader.GetOrdinal("maintenance_date")))
254 {
255 try
256 {
257 string rawDate = reader.GetValue(reader.GetOrdinal("maintenance_date")).ToString();
258
259 if (DateTime.TryParse(rawDate, out DateTime parsedDate))
260 task.MaintenanceDate = parsedDate.ToString("yyyy-MM-dd HH:mm:ss");
261 else
262 task.MaintenanceDate = "Invalid Date";
263 }
264 catch (Exception ex)
265 {
266 Console.WriteLine($"Error parsing date: {ex.Message}");
267 task.MaintenanceDate = "Invalid Date";
268 }
269 }
270 else
271 {
272 task.MaintenanceDate = "No Date";
273 }
274
275 // Get the associated user's email
276 int maintainerId = reader.GetInt32(reader.GetOrdinal("maintainer_id"));
277 task.MaintainerEmail = GetUserEmailById(maintainerId);
278
279 task.Timestamp = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss");
280 maintenanceTasks.Add(task);
281 }
282 }
283 }
284
285 return maintenanceTasks;
286 }
